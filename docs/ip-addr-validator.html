#Validates if a string is a valid IPv4 address.

import re

def is_valid_ip(ip):
    ip_regex = r'^((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$'
    return bool(re.match(ip_regex, ip))

# Test cases

ips = [
    "192.168.1.1",
    "255.255.255.255",
    "0.0.0.0",
    "300.168.1.1",
    "abc.def.ghi.jkl"
]

#Print valid 

for ip in ips:
    print(f"{ip}: {'Valid' if is_valid_ip(ip) else 'Invalid'}")
