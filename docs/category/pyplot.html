#Adding annotations

import matplotlib.pyplot as plt

x = [1, 2, 3, 4, 5]

y = [1, 4, 9, 16, 25]

plt.plot(x, y, marker="o")
for i, j in zip(x, y):
    plt.text(i, j+0.5, f"({i}, {j})")
