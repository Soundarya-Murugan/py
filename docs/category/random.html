#Dice Roll Simulation

import random

def roll_dice():
    return random.randint(1, 6)

def simulate_dice_rolls(rolls=100):
    results = {i: 0 for i in range(1, 7)}
    for _ in range(rolls):
        roll = roll_dice()
        results[roll] += 1
    return results

rolls = 1000

results = simulate_dice_rolls(rolls)

print(f"Dice Roll Simulation Results (Total Rolls: {rolls})")

Dice Roll Simulation Results (Total Rolls: 1000)

for num, count in results.items():
    print(f"{num}: {count}")

#Lottery Number Generator

import random

def generate_lottery_numbers():
    main_numbers = random.sample(range(1, 50), 5)
    powerball = random.randint(1, 20)
    return main_numbers, powerball

print("Lottery Numbers:")
for i in range(3):
    main, powerball = generate_lottery_numbers()
    print(f"Main Numbers: {main}, Powerball: {powerball}")
