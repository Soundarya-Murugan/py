#Validates a date format (e.g., DD/MM/YYYY).

import re

def is_valid_date(date):
    date_regex = r'^(0[1-9]|[12][0-9]|3[01])/(0[1-9]|1[0-2])/\d{4}$'
    return bool(re.match(date_regex, date))

# Test cases
dates = [
    "31/12/2020",
    "01/01/2000",
    "29/02/2021",
    "32/01/2021",
    "12/13/2021"
]

#Print the date

for date in dates:
    print(f"{date}: {'Valid' if is_valid_date(date) else 'Invalid'}")

import re

def replace_words(text, word_dict):
    for word, replacement in word_dict.items():
        pattern = r'\b' + re.escape(word) + r'\b'
        text = re.sub(pattern, replacement, text)
    return text

text = "I love Python. Python is awesome."

word_dict = {"Python": "Java"}

modified_text = replace_words(text, word_dict)
