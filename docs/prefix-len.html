import namegenerator

def generate_names_with_prefix(prefix, min_length):
    name = namegenerator.gen(prefix)
    while len(name) < min_length:
        name = namegenerator.gen(prefix)
    return name

prefix = "super"

min_length = 8

name = generate_names_with_prefix(prefix, min_length)

print(f"Generated name: {name}")
