import re

def replace_words(text, word_dict):
    for word, replacement in word_dict.items():
        pattern = r'\b' + re.escape(word) + r'\b'
        text = re.sub(pattern, replacement, text)
    return text

text = "I love Python. Python is awesome."

word_dict = {"Python": "Java"}

modified_text = replace_words(text, word_dict)

print("Modified text:", modified_text)

import namegenerator

def generate_and_reverse_name():
    name = namegenerator.gen()
    reversed_name = name[::-1]
    return reversed_name

reversed_name = generate_and_reverse_name()
print(f"Original name: {namegenerator.gen()}")

import re

def is_valid_ean13(ean):
    return bool(re.match(r'^\d{13}$', ean))

ean_input = '123456789012'

if is_valid_ean13(ean_input):
    print(f'{ean_input} is a valid EAN-13')
