#Table with Matplotlib

import matplotlib.pyplot as plt

# Data for table

data = [['Alice', 85, 'A'],
        ['Bob', 78, 'B'],
        ['Charlie', 92, 'A+']]

column_labels = ('Name', 'Score', 'Grade')

# Plot

fig, ax = plt.subplots()
ax.axis('tight')
ax.axis('off')
ax.table(cellText=data, colLabels=column_labels, loc='center')

plt.title("Table Example")
plt.show()
