import namegenerator

def generate_unique_names(num):
    unique_names = set()
    while len(unique_names) < num:
        name = namegenerator.gen()
        if name not in unique_names:
            unique_names.add(name)
    return list(unique_names)
names = generate_unique_names(5)
print("Generated unique names:")
for name in names:
    print(name)

import magicattr

def has_attr(obj, attr):
    try:
        magicattr.get(obj, attr)
        return True
    except AttributeError:
        return False

class Example:
    def __init__(self):
        self.a = {'b': {'c': 'value'}}

obj = Example()

# Check if nested attributes exist
print(has_attr(obj, 'a.b.c'))  # Output: True
print(has_attr(obj, 'a.b.d'))  # Output: False …
