import magicattr

class Example:
    def __init__(self):
        self.name = 'example'
        self.version = 1.0
        self.description = 'A test example'

# Create an instance of Example
obj = Example()

# Accessing multiple attributes
attrs = [magicattr.get(obj, attr) for attr in ['name', 'version', 'description']]
print(attrs)  # Output: ['example', 1.0, 'A test example']
