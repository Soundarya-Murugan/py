from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

age

import magicattr

def has_attr(obj, attr):
    try:
        magicattr.get(obj, attr)
        return True
    except AttributeError:
        return False

class Example:
    def __init__(self):
        self.a = {'b': {'c': 'value'}}

obj = Example()

# Check if nested attributes exist
print(has_attr(obj, 'a.b.c'))  # Output: True
print(has_attr(obj, 'a.b.d'))  # Output: False …
